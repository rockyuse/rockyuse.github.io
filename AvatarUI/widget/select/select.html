模拟下拉框
默认 自定义宽度 自定义下拉框的高度 自定义下拉框的高度根据条目来定 已选择项的id 已选择项的名称 初始化后的回调 勾选后的回调 反勾选后的回调 单选 显示全选 全选后的回调 反全选后的回调 显示选择提示框 自定义选择提示框宽度

默认


		            $("#Select1").ImitSelect({
						option: opts
					});

自定义宽度


		            $("#Select2").ImitSelect({
						selectWidth: 250,
						selectUlWidth: 300,
						option: opts
					});

自定义下拉框的高度


		            $("#Select3").ImitSelect({
						selectUlHeight: "300",
						option: opts
					});

自定义下拉框的高度根据条目来定


		            $("#Select4").ImitSelect({
						selectUlNum: 7,
						option: opts
					});

已选择项的id


		            $("#Select5").ImitSelect({
						selectOptionId: [1,2,13],
						option: opts
					});

已选择项的名称


		            $("#Select6").ImitSelect({
						selectOptionName: ["中国", "美国", "英国"],
						option: opts
					});

初始化后的回调


		            $("#Select7").ImitSelect({
						initCallback: function(){$('#SelectInitSuccess').html("初始化成功！");},
						option: opts
					});

勾选后的回调


		            $("#Select8").ImitSelect({
						selectCallback: function(id, name){alert('选择项的id是'+id+', name是'+name);},
						option: opts
					});

反勾选后的回调


		            $("#Select9").ImitSelect({
						removeCallback: function(id, name){alert('移除项的id是'+id+', name是'+name);},
						option: opts
					});

单选


		            $("#Select10").ImitSelect({
						selectType: "single",
						option: opts
					});

显示全选


		            $("#Select11").ImitSelect({
						showSelectAll: true,
						option: opts
					});

全选后的回调


		            $("#Select12").ImitSelect({
						showSelectAll: true,
						selectAllCallback: function(){alert("全选了！")},
						option: opts
					});

反全选后的回调


		            $("#Select13").ImitSelect({
						showSelectAll: true,
						unSelectAllCallback: function(){alert("反全选了！")},
						option: opts
					});

显示选择提示框


		            $("#Select14").ImitSelect({
						showTip: true,
						option: opts
					});

自定义选择提示框宽度


		            $("#Select15").ImitSelect({
						showTip: true,
						selectTipWidth: 200,
						option: opts
					});
