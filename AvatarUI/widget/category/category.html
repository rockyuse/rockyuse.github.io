无限级分类
默认_ 逐步请求分类 一次性加载所有分类 静态的分类数据 已有数据的转换(value) 已有数据的转换(name) 已有数据的转换(name)--自定义分隔符

默认_ 逐步请求分类

		            //默认_ 逐步请求分类
					$("#Category_0").Category({
						url: "../../category/getCategory.do"
					});

一次性加载所有分类

		            //一次性加载所有分类
					$("#Category_1").Category({
						url: "../../category/getCategory.do",
						getData: "once"
					});

静态的分类数据

					//静态的分类数据
					$("#Category_2").Category({
						getData: "once",
						staticData: '{"category":[{"id":"1","fid":"0","name":"家用电器"},{"id":"2","fid":"0","name":"服饰"},{"id":"3","fid":"0","name":"化妆"},{"id":"4","fid":"1","name":"大家电"},{"id":"5","fid":"1","name":"生活电器"},{"id":"6","fid":"1","name":"厨房电器"},{"id":"7","fid":"4","name":"空调"},{"id":"8","fid":"4","name":"冰箱"},{"id":"9","fid":"4","name":"洗衣机"},{"id":"10","fid":"4","name":"热水器"},{"id":"11","fid":"3","name":"面部护理"},{"id":"12","fid":"3","name":"口腔护理"},{"id":"13","fid":"2","name":"男装"},{"id":"14","fid":"2","name":"女装"},{"id":"15","fid":"7","name":"海尔空调"},{"id":"16","fid":"7","name":"美的空调"},{"id":"19","fid":"5","name":"加湿器"},{"id":"20","fid":"5","name":"电熨斗"}]}'

已有数据的转换(value)

hasValue: "1-4-7"

					//已有数据的转换(value)
					$("#Category_3").Category({
						url: "../../category/getCategory.do",
						getData: "once",
						hasValue: "1-4-7"
					});

已有数据的转换(name)

hasName: "家用电器-大家电-空调-海尔空调";

					//已有数据的转换(name)
					$("#Category_4").Category({
						url: "../../category/getCategory.do",
						getData: "once",
						hasName: "家用电器-大家电-空调-海尔空调"
					});

已有数据的转换(name)--自定义分隔符

hasName: "家用电器||大家电||空调||海尔空调";

					//已有数据的转换(name)--自定义分隔符
					$("#Category_5").Category({
						url: "../../category/getCategory.do",
						getData: "once",
						hasName: "家用电器||大家电||空调||海尔空调",
						splitStr: "||"
					});
