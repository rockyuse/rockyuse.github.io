自定义回调

	            	$('#Pie03').Pie({
						width: 200,
						height: 200,
						radius: 58,
						offsetX: 20,
						offsetY: 20,
						bgColor: '#fff',
						colorArr: ['#52A8E5', '#FE8385'],
						valueArr: [12.5, 87.5],
						startAngle: -(12.5*3.6/2) - 45,

						circleInnerRadius: 44,
						circleInnerColor: '#fff',

						beforeCallback: function(thisCanvas){
							thisCanvas.image("images/male.png", 108, 8, 43, 42);
							thisCanvas.image("images/female.png", 9, 107, 41, 42);
						},

						initCallback: function(thisCanvas){
							thisCanvas.image("images/person.png", 59, 60, 43, 41);
						}
					});
