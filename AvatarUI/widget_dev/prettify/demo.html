//格式化数字
function formatNum(str){
	str = str.toString();
	if(/[^0-9\.]/.test(str)){return str;}
	var strFloor = '';
	if(RegExp('\\.').test(str)){
		strArr = str.split('.');
		str = strArr[0];
		strFloor = '.' + strArr[1];
	}
	var n = str.length % 3;
	if(n){
		return str.slice(0,n) +  str.slice(n).replace(/(\d{3})/g,',$1') + strFloor;
	}else{
		return str.replace(/(\d{3})/g,',$1').slice(1) + strFloor;
	}
}
