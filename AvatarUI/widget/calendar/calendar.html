日历
默认单日期 多月份 日历框偏移 初始化日历框后执行回调 选择日期后执行回调 当天日期自定义 选择日期段 显示快速选择 显示时分秒 显示时分秒-变化的 显示小时 显示小时--小时自定义 禁止选择过去日期 禁止选择将来日期 禁止选择日期段之外的日期 显示对比--默认同比 显示对比--环比

默认单日期





		            $('#Calendar1').Calendar();

多月份

                    $('#Calendar2').Calendar({
                        monthSize: 3
                    });

日历框偏移

                    $('#Calendar3').Calendar({
                        offsetX: 10,
						offsetY: 20
                    });

初始化日历框后执行回调

					$('#Calendar4').Calendar({
                        initCallback: function(){
							alert('日历框初始化完毕！');
						}
                    });

选择日期后执行回调

                    $('#Calendar5').Calendar({
                        callback: function(){
                            alert('执行回调！');
                        }
                    });

当天日期自定义

					var currentDate = new Date();
					currentDate.setFullYear(2009);
					currentDate.setMonth(5);
					$('#Calendar6').Calendar({
						currentDate: currentDate
					});

选择日期段

					$('#Calendar7').Calendar({
						single: false,
						monthSize: 3
					});

显示快速选择

					$('#Calendar8').Calendar({
						single: false,
						monthSize: 3
						showShortcuts: true
					});

显示时分秒

					$('#Calendar9').Calendar({
						showTime: true
					});

显示时分秒-变化的

					$('#Calendar10').Calendar({
						showTime: "current"
					});

显示小时

					$('#Calendar11').Calendar({
						showHour: true
					});

显示小时--小时自定义

					$('#Calendar12').Calendar({
						showHour: "5"
					});

禁止选择过去日期

					$('#Calendar13').Calendar({
						disableDate: "past"
					});

禁止选择将来日期

					$('#Calendar14').Calendar({
						disableDate: "future"
					});

禁止选择日期段之外的日期

					$('#Calendar15').Calendar({
						disableDate: "2012-01-05||2012-01-20"
					});

显示对比--默认同比

					$('#Calendar16').Calendar({
						showCompare: true,
						single: false,
						monthSize: 3,
						compareId: "CalendarCompare_16"
					});

显示对比--环比

					$('#Calendar17').Calendar({
						single: false,
						monthSize: 3,
						showCompare: true,
						compareType: "huanbi",
						compareId: "CalendarCompare_17"
					});
